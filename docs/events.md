# Events

Events are delivered to your inline handlers or `update()` function.
Each event has a `kind` discriminant field for type narrowing via
type guards.

## Widget events

Generated by user interaction with widgets. Widget events have
`kind: "widget"` and carry `id`, `scope`, `type`, `value`, and
`data` fields.

### Click

```typescript
// event shape:
// { kind: "widget", type: "click", id: "save", scope: [], value: null, data: null }
```

Generated by `Button` widgets when pressed.

```typescript
// Inline handler (preferred):
<Button id="save" onClick={(s) => save(s)}>Save</Button>

// Or in update():
function update(state: Model, event: Event): Model {
  if (isClick(event, 'save')) return save(state)
  if (isClick(event, 'cancel')) return revert(state)
  return state
}
```

### Input

```typescript
// event shape:
// { kind: "widget", type: "input", id: "search", value: "query text", ... }
```

Generated by `TextInput` on every keystroke (when `onInput` is set,
which is the default). `event.value` is the current text content as
a string.

```typescript
<TextInput id="search" value={state.query}
  onInput={(s, e) => ({ ...s, query: e.value as string })} />

// In update():
if (isInput(event, 'search')) {
  return { ...state, searchQuery: event.value as string }
}
```

### Submit

```typescript
// event shape:
// { kind: "widget", type: "submit", id: "search", value: "query text", ... }
```

Generated by `TextInput` when the user presses Enter (when `onSubmit`
is set).

```typescript
<TextInput id="search" value={state.query}
  onSubmit={(s) => [s, searchCommand(s.query)]} />

// In update():
if (isSubmit(event, 'search')) {
  return [{ ...state, searchQuery: event.value as string }, searchCommand(event.value as string)]
```


### Slide and slide release

```typescript
// event shapes:
// { kind: "widget", type: "slide", id: "volume", value: 75, ... }
// { kind: "widget", type: "slide_release", id: "volume", value: 75, ... }
```

Generated by `Slider` and `VerticalSlider`. `slide` fires
continuously during dragging. `slide_release` fires once when the
user releases the slider.

```typescript
<Slider id="volume" value={state.volume} range={[0, 100]}
  onSlide={(s, e) => ({ ...s, volume: e.value as number })}
  onSlideRelease={(s) => [s, Command.async(async () => savePreference('volume', s.volume), 'saved')]} />

// In update():
if (isSlide(event, 'volume')) {
  return { ...state, volume: event.value as number }
}
// isWidget(event) && event.type === 'slide_release' for release events
if (isWidget(event) && event.type === 'slide_release' && event.id === 'volume') {
  return [{ ...state, volume: event.value as number }, savePreference('volume', event.value)]
}
```

### Scroll

```typescript
// event shape:
// {
//   kind: "widget", type: "scroll", id: "log_view",
//   data: {
//     absolute_x: number,
//     absolute_y: number,
//     relative_x: number,   // 0.0 - 1.0
//     relative_y: number,   // 0.0 - 1.0
//     bounds: [width, height],
//     content_bounds: [width, height],
//   }

```

Generated by `Scrollable` when the scroll position changes (when
`onScroll` is set). The viewport data includes both absolute pixel
offsets and relative (0.0-1.0) scroll positions, plus the visible

```typescript
if (isWidget(event) && event.type === 'scroll' && event.id === 'logView') {
  const viewport = event.data as Record<string, unknown>
  const atBottom = (viewport['relative_y'] as number) >= 0.99
  return { ...state, autoScroll: atBottom }
}
```

## Mouse area events

Mouse area events are widget events. Each event requires its
corresponding boolean prop to be set on the `MouseArea` widget.
Without the prop, the event is not emitted.

Available types: `mouse_right_press`, `mouse_right_release`,
`mouse_middle_press`, `mouse_middle_release`, `mouse_double_click`,
`mouse_enter`, `mouse_exit`, `mouse_move`, `mouse_scroll`.

Note: left press/release events from mouse areas are delivered as
`type: "click"` widget events.

```typescript
// event shapes:
// { kind: "widget", type: "mouse_right_press", id: "canvas", ... }
// { kind: "widget", type: "mouse_enter", id: "tooltip_target", ... }
// { kind: "widget", type: "mouse_move", id: "drag_zone", data: { x: 150, y: 200 }, ... }
// { kind: "widget", type: "mouse_scroll", id: "scroll_zone", data: { delta_x: 0, delta_y: -3 }, ... }
```

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isMouseArea(event, 'hoverZone') && event.type === 'mouse_enter') {
    return { ...state, hovered: true }
  }
  if (isMouseArea(event, 'hoverZone') && event.type === 'mouse_exit') {
    return { ...state, hovered: false }
  }
  if (isMouseArea(event, 'canvasArea') && event.type === 'mouse_move') {
    const d = event.data as Record<string, unknown>
    return { ...state, cursor: { x: d['x'] as number, y: d['y'] as number } }
  }
  if (isMouseArea(event, 'editor') && event.type === 'mouse_right_press') {
    return { ...state, contextMenuOpen: true }
  }
  if (isMouseArea(event, 'label') && event.type === 'mouse_double_click') {
    return { ...state, editing: true }
  }
  if (isMouseArea(event, 'zoomArea') && event.type === 'mouse_scroll') {
    const d = event.data as Record<string, unknown>
    const deltaY = d['delta_y'] as number
    return { ...state, zoom: state.zoom + deltaY * 0.1 }
  }
  return state
}
```

## Canvas events

Generated by `Canvas` widgets. Each event is opt-in via a boolean prop
on the canvas node.

```typescript
// event shapes:
// { kind: "widget", type: "canvas_press", id: "draw_area", data: { x: 50, y: 100, button: "left" }, ... }
// { kind: "widget", type: "canvas_release", id: "draw_area", data: { x: 50, y: 100, button: "left" }, ... }
// { kind: "widget", type: "canvas_move", id: "draw_area", data: { x: 75, y: 125 }, ... }
// { kind: "widget", type: "canvas_scroll", id: "draw_area", data: { x: 75, y: 125, delta_x: 0, delta_y: -3 }, ... }
```

The `button` field is a string (`"left"`, `"right"`, `"middle"`). The
`x`/`y` coordinates are relative to the canvas origin. All fields live
in the `data` map.

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isCanvas(event, 'drawArea') && event.type === 'canvas_press') {
    const d = event.data as Record<string, unknown>
    if (d['button'] === 'left') {
      return { ...state, drawing: true, lastPoint: { x: d['x'] as number, y: d['y'] as number } }
    }
  }
  if (isCanvas(event, 'drawArea') && event.type === 'canvas_move') {
    if (state.drawing) {
      const d = event.data as Record<string, unknown>
      return {
        ...state,
        lastPoint: { x: d['x'] as number, y: d['y'] as number },
        strokes: [...state.strokes, { x: d['x'] as number, y: d['y'] as number }],
      }
    }
  }
  if (isCanvas(event, 'drawArea') && event.type === 'canvas_release') {
    return { ...state, drawing: false }
  }
  return state
}
```

## Canvas element events

When a canvas contains shapes with an `interactive` field (see
[composition patterns](composition-patterns.md#canvas-interactive-shapes)),
the renderer handles hit testing locally and emits semantic shape
events. These arrive as `kind: "widget"` events with `canvas_element_*` types.
The `id` is the canvas widget ID; `data.element_id` identifies the
specific shape.

```typescript
// event shapes:
// { kind: "widget", type: "canvas_element_enter", id: "chart",
//   data: { element_id: "bar-jan", x: 15.0, y: 70.0 } }
//
// { kind: "widget", type: "canvas_element_leave", id: "chart",
//   data: { element_id: "bar-jan" } }
//
// { kind: "widget", type: "canvas_element_click", id: "chart",
//   data: { element_id: "bar-jan", x: 15.0, y: 70.0, button: "left" } }
//
// { kind: "widget", type: "canvas_element_drag", id: "chart",
//   data: { element_id: "handle", x: 50.0, y: 80.0, delta_x: 2.0, delta_y: -1.0 } }
//
// { kind: "widget", type: "canvas_element_drag_end", id: "chart",
//   data: { element_id: "handle", x: 52.0, y: 79.0 } }
//
// { kind: "widget", type: "canvas_element_focused", id: "chart",
//   data: { element_id: "bar-jan" } }
```

Hover styles, pressed styles, cursors, and tooltips on shapes are
handled by the renderer locally -- no round-trip needed. Shape events
give the host semantic actions (clicks, drags, focus changes) instead
of raw coordinates.

```typescript
if (isWidget(event) && event.type === 'canvas_element_click' && event.id === 'chart') {
  const d = event.data as Record<string, unknown>
  return { ...state, selectedBar: d['element_id'] as string }
}

if (isWidget(event) && event.type === 'canvas_element_drag' && event.id === 'chart') {
  const d = event.data as Record<string, unknown>
  return {
    ...state,
    handlePos: {
      x: d['x'] as number,
      y: d['y'] as number,
    },
  }
}
```

## Keyboard events

Delivered when keyboard subscriptions are active (see
[commands.md](commands.md)).

```typescript
// event shapes:
// { kind: "key", type: "press", key: "s", modifiers: { command: true, ... }, ... }
// { kind: "key", type: "release", key: "Escape", ... }
```

### KeyEvent fields

```typescript
interface KeyEvent {
  kind: "key"
  type: "press" | "release"
  key: string                   // logical key (named key or character)
  modifiedKey: string | null    // key with modifiers applied (e.g. Shift+a = "A")
  physicalKey: string | null    // physical key code (layout-independent)
  location: "standard" | "left" | "right"
  modifiers: Modifiers
  text: string | null           // text produced by the key press (null for non-printable)
  repeat: boolean               // true if this is an auto-repeat event
  captured: boolean
  tag: string
}
```

### Modifiers

```typescript
interface Modifiers {
  shift: boolean
  ctrl: boolean
  alt: boolean
  logo: boolean
  command: boolean   // platform-aware: ctrl on Linux/Windows, logo (Cmd) on macOS
}
```

### Key values

Named keys are strings matching the key name:

```
"Enter", "Escape", "Tab", "Backspace", "Delete",
"ArrowUp", "ArrowDown", "ArrowLeft", "ArrowRight",
"Home", "End", "PageUp", "PageDown",
"F1", "F2", ... "F12",
"Space"
```

Character keys are single-character strings:

```
"a", "b", "1", "/", " "
```

### Keyboard event examples

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isKey(event, 'press')) {
    // Ctrl+S / Cmd+S to save
    if (event.key === 's' && event.modifiers.command) {
      return [state, saveCommand()]
    }

    // Escape to close modal
    if (event.key === 'Escape') {
      return { ...state, modalOpen: false }
    }

    // Ctrl+Z to undo, Ctrl+Shift+Z to redo
    if (event.key === 'z' && event.modifiers.command && !event.modifiers.shift) {
      return undo(state)
    }
    if (event.key === 'z' && event.modifiers.command && event.modifiers.shift) {
      return redo(state)
    }

    // Use physicalKey for layout-independent bindings (e.g. WASD on non-QWERTY)
    if (event.physicalKey === 'KeyW') {
      return moveUp(state)
    }

    // Use text field for text input handling
    if (event.text != null) {
      return appendChar(state, event.text)
    }
  }

  return state
}
```

## IME events

Delivered when IME subscriptions are active (see
[commands.md](commands.md)). Input Method Editor events support CJK
and other compose-based text input.

```typescript
// event shapes:
// { kind: "ime", type: "opened", ... }
// { kind: "ime", type: "preedit", text: "composition text", cursor: [start, end] | null, ... }
// { kind: "ime", type: "commit", text: "final text", ... }
// { kind: "ime", type: "closed", ... }
```

`preedit` fires during composition with the in-progress text and
optional cursor range. `commit` fires when the user finalises a
composed string.

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isIme(event)) {
    switch (event.type) {
      case 'preedit':
        return { ...state, composing: event.text }
      case 'commit':
        return { ...state, composing: null, value: state.value + (event.text ?? '') }
    }
  }
  return state
}
```

## Mouse events (global)

Delivered when mouse subscriptions are active. These are global
(not widget-scoped) events from the windowing system.

```typescript
// event shapes:
// { kind: "mouse", type: "moved", x: 150, y: 200, ... }
// { kind: "mouse", type: "entered", ... }
// { kind: "mouse", type: "left", ... }
// { kind: "mouse", type: "pressed", button: "left", ... }
// { kind: "mouse", type: "released", button: "right", ... }
// { kind: "mouse", type: "scrolled", deltaX: 0, deltaY: -3, ... }
```

The `button` field is a string (`"left"`, `"right"`, `"middle"`,
`"back"`, `"forward"`). Scroll events include `deltaX`/`deltaY`.

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isMouse(event)) {
    switch (event.type) {
      case 'moved':
        return { ...state, cursor: { x: event.x, y: event.y } }
      case 'pressed':
        if (event.button === 'left') return { ...state, mouseDown: true }
        break
      case 'released':
        if (event.button === 'left') return { ...state, mouseDown: false }
        break
      case 'scrolled':
        return { ...state, scrollOffset: state.scrollOffset + event.deltaY }
    }
  }
  return state
}
```

## Touch events

Delivered when touch subscriptions are active. Each touch point is
tracked by `fingerId`.

```typescript
// event shapes:
// { kind: "touch", type: "pressed", fingerId: 0, x: 100, y: 200, ... }
// { kind: "touch", type: "moved", fingerId: 0, x: 105, y: 210, ... }
// { kind: "touch", type: "lifted", fingerId: 0, x: 105, y: 210, ... }
// { kind: "touch", type: "lost", fingerId: 0, x: 105, y: 210, ... }
```

The `fingerId` uniquely identifies a touch point across its lifetime
(pressed -> moved -> lifted/lost). Track multiple fingers to
implement pinch-to-zoom or multi-touch gestures.

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isTouch(event)) {
    switch (event.type) {
      case 'pressed':
        return { ...state, touchStart: { x: event.x, y: event.y }, activeFinger: event.fingerId }
      case 'moved':
        if (event.fingerId === state.activeFinger) {
          return { ...state, touchPos: { x: event.x, y: event.y } }
        }
        break
      case 'lifted':
        return { ...state, activeFinger: null }
    }
  }
  return state
}
```

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isWindow(event) && event.type === 'close_requested' && event.windowId === 'main') {
    if (state.unsavedChanges) {
```

### File drag and drop

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isWindow(event) && event.windowId === 'main') {
    switch (event.type) {
      case 'file_hovered': {
        const d = event.data as Record<string, unknown>
        return { ...state, dropTargetActive: true, hoveredFile: d['path'] as string }

    }
  }
  return state
}
```

## System events

System events have `kind: "system"`. The `type` field identifies the
specific system event. The `data` field carries the payload.

```typescript
// event shapes:
// { kind: "system", type: "animation_frame", data: 16843521, tag: "frame" }
// { kind: "system", type: "theme_changed", data: "dark", tag: "theme" }
// { kind: "system", type: "all_windows_closed", data: null, tag: "" }
```

Animation frame events are delivered on each frame when an animation
subscription is active. `data` holds a monotonic timestamp in
milliseconds. Theme changed events fire when the OS theme switches.
`data` is `"light"` or `"dark"`.

```typescript
if (isSystem(event, 'animation_frame')) {
  return { ...state, lastFrame: event.data as number }
}

if (isSystem(event, 'theme_changed')) {
  return { ...state, osTheme: event.data as string }
}

if (isSystem(event, 'all_windows_closed')) {
  return [state, Command.exit()]
}
```

## Async events (command results)

Delivered when a `Command.async()` completes.

<!-- test: events_async_result_ok_guard, events_async_result_error_guard -- keep this code block in sync with the test -->
```typescript
// event shapes:
// { kind: "async", tag: "dataLoaded", result: { ok: true, value: responseData } }
// { kind: "async", tag: "dataLoaded", result: { ok: false, error: errorInfo } }
```

Where `tag` is the string you passed to `Command.async()`.

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isClick(event, 'fetch')) {
    return [
      { ...state, loading: true },
      Command.async(async () => {
        const res = await fetch('/api/data')
        return res.json()
      }, 'dataLoaded'),
    ]
  }

  if (isAsync(event, 'dataLoaded')) {
    if (event.result.ok) {
      return { ...state, loading: false, data: event.result.value }
    }
    return { ...state, loading: false, error: String(event.result.error) }
  }

  return state
}
```

## Effect events

Delivered when a renderer effect (file dialog, clipboard, notification)
completes.

<!-- test: events_effect_ok_guard, events_effect_cancelled_guard -- keep this code block in sync with the test -->
```typescript
// event shapes:
// { kind: "effect", requestId: "ef_1234", status: "ok", result: "/path/to/file", error: null }
// { kind: "effect", requestId: "ef_1234", status: "cancelled", result: null, error: null }
// { kind: "effect", requestId: "ef_1234", status: "error", result: null, error: "permission denied" }
```

The `requestId` is the auto-generated effect ID (e.g. `"ef_1234"`),
not the effect kind name. The `status` field is one of `"ok"`,
`"cancelled"`, or `"error"`.

```typescript
function update(state: Model, event: Event): UpdateResult<Model> {
  if (isEffect(event)) {
    switch (event.status) {
      case 'ok':
        return [state, loadFile(event.result as string)]
      case 'cancelled':
        return state
      case 'error':
        return { ...state, error: event.error }
    }
  }
  return state
}
```

See [effects.md](effects.md).

## Pattern matching tips

TypeScript doesn't have Elixir's pattern matching, but type guards
and destructuring cover the same ground.

### Matching by widget ID prefix

<!-- test: events_pattern_prefix_match -- keep this code block in sync with the test -->
```typescript
if (isClick(event)) {
  // Match clicks with a prefix
  if (event.id.startsWith('nav:')) {
    const section = event.id.slice(4)
    return { ...state, section }
  }

  // Match toggle on any "setting:" prefixed checkbox
  if (event.type === 'toggle' && event.id.startsWith('setting:')) {
    const key = event.id.slice(8)
    return { ...state, settings: { ...state.settings, [key]: event.value } }
  }
}
```

### Scope matching

Widget events include a `scope` array listing ancestor container IDs,
nearest first. Match on scope to distinguish events from different
contexts:

<!-- test: events_scope_sidebar_match, events_scope_main_match -- keep this code block in sync with the test -->
```typescript
if (isClick(event, 'save')) {
  // Button inside the "sidebar" container
  if (event.scope[0] === 'sidebar') {
    return saveSidebar(state)
  }
  // Same button ID but inside "main" container
  if (event.scope[0] === 'main') {
    return saveMain(state)
  }
}
```

### Reconstructing the full path

Use `target()` to reconstruct the full scoped ID path from an event:

<!-- test: events_target_reconstruction, events_target_no_scope -- keep this code block in sync with the test -->
```typescript
import { target } from 'plushie'

// { id: "save", scope: ["form", "app"] } -> "app/form/save"
const fullPath = target(event as WidgetEvent)
```

### Catch-all

Always include a catch-all return in your `update()`:

<!-- test: events_catch_all -- keep this code block in sync with the test -->
```typescript
function update(state: Model, event: Event): Model {
  // ... event handling ...
  return state
}
```

Unknown events are silently ignored. This is important for forward
compatibility -- new widget types or renderer versions may emit
events your app does not yet handle.
